Let's put that into practice


                public interface ThrowingFunction<T, R> {
                    R apply(T t) throws Exception;
                }
            

                static <T, R> Function<T, R> sneaky(ThrowingFunction<T, R> f)
            

                static <T, R> Function<T, R> sneaky(ThrowingFunction<T, R> f) {
                    return t -> {
                        try {
                            return f.apply(t);
                        } catch (Exception ex) {
                            return ThrowingFunction.sneakyThrow(ex);
                        }
                    };
                 }
