		go func(url string) {
			content, err := fetchURL(url)
			if err != nil {
				results <- fmt.Sprintf("Error fetching %s: %v", url, err)
				return
			}
			results <- fmt.Sprintf("Fetched %s: %s...", url, content)
		}(url)
	}

	for range urls {

	}
